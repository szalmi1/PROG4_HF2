242  if (item.Completed)
243  {
244  while (this.ArchivedRentals.ContainsKey(item.End))

246  item.End.AddMilliseconds(1);
247  }
248 
249  this.ArchivedRentals.Add(item.End, this.RentalRepo.GetOne(archivedRentalId).ToString());

251  }
